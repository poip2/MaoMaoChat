## 第4步：openFile 函数（核心）

**文件**: `src/lib/tauri/files.ts:17-55`
```javascript
export async function openFile(path: string): Promise<void> {
    const fileName = path.split("/").pop() ?? path;

    // ① 设置加载状态
    document.set({
        filePath: path,
        fileName,
        content: "",
        renderedHtml: "",
        loading: true,        // ← 显示"加载中..."
        error: null,
    });

    try {
        // ② 调用 Rust 读取文件
        const content = await readMarkdownFile(path);  // ← 关键调用
        
        // ③ 渲染 Markdown 为 HTML
        const result = renderFull(content, baseDir);
        
        // ④ 创建新标签页
        tabStore.addTab(path, fileName, content, result.html, ...);
        
        // ⑤ 更新显示状态
        document.set({
            ...result,
            loading: false,   // ← 停止加载，显示内容
        });

        // ⑥ 添加到最近文件
        addRecentFile(path, fileName);
        
        // ⑦ 开始监视文件变化
        invoke("start_watching", { path });
        
    } catch (err) {
        document.set({ error: `Failed to open: ${err}` });
    }
}
```

```rust
#[tauri::command]
pub fn read_markdown_file(path: String) -> Result<String, String> {
    let p = Path::new(&path);

    // 检查文件是否存在
    if !p.exists() {
        return Err(format!("File not found: {}", path));
    }

    // 检查是否是文件（不是目录）
    if !p.is_file() {
        return Err(format!("Not a file: {}", path));
    }

    // 读取文件内容并返回
    fs::read_to_string(p).map_err(|e| format!("Failed to read: {}", e))
}
```

**返回**: 文件内容字符串 → 通过 Tauri IPC → 回到前端

## 第7步：渲染 Markdown

**文件**: `src/lib/renderer/pipeline.ts:88-130`
```javascript
export function renderFull(markdown: string, baseDir?: string): RenderResult {
    // ① 提取前置元数据 (YAML frontmatter)
    let frontmatter = 解析YAML(markdown);
    let content = 移除frontmatter部分(markdown);
    
    // ② 统计字数
    let wordCount = content.trim().split(/\s+/).length;
    
    // ③ Markdown → HTML (使用 markdown-it)
    let raw = md.render(content);
    
    // ④ 安全过滤 (DOMPurify)
    let html = DOMPurify.sanitize(raw);
    
    // ⑤ 解析相对路径图片
    html = resolveRelativeImages(html, baseDir);
    
    return { html, frontmatter, wordCount };
}
```

**使用的库**:
- `markdown-it` - Markdown 解析
- `highlight.js` - 代码高亮
- `KaTeX` - 数学公式
- `DOMPurify` - XSS 防护

## 第8步：创建标签页

**文件**: `src/lib/stores/tabs.ts:31-58`
```javascript
function addTab(filePath, fileName, content, renderedHtml, ...) {
    const currentTabs = get(tabs);

    // 如果文件已打开，切换到该标签
    const existing = currentTabs.find((t) => t.filePath === filePath);
    if (existing) {
        activeTabId.set(existing.id);  // 切换
        return existing.id;
    }

    // 否则创建新标签
    const id = generateId();
    const newTab = {
        id,
        filePath,
        fileName,
        content,           // 原始 Markdown
        renderedHtml,      // 渲染后的 HTML
        isEditing: false,
        dirty: false,
    };

    tabs.update((ts) => [...ts, newTab]);  // 添加到列表
    activeTabId.set(id);                   // 设为活动标签
    return id;
}
```
