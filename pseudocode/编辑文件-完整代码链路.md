```javascript
function switchMode(target: ViewMode) {
    if (!activeTab || target === currentMode) return;
    
    const line = getCurrentSourceLine(currentMode);  // 保存当前位置

    if (target === "editor") {
        // ★ 关键：设置标签为编辑状态
        tabStore.setEditing(activeTab.id, true);
    }
    
    // 滚动同步（编辑器跳到对应位置）
    tick().then(() => {
        requestAnimationFrame(() => {
            scrollToSourceLine(target, line);
        });
    });
}
```

## 第4步：setEditing 更新标签状态

**文件**: `src/lib/stores/tabs.ts:147-158`
```javascript
function setEditing(id: string, editing: boolean) {
    tabs.update((ts) =>
        ts.map((t) => {
            if (t.id !== id) return t;
            
            // 进入编辑模式时，同步 editContent 为当前内容
            if (editing && !t.isEditing && !t.dirty) {
                return { 
                    ...t, 
                    isEditing: true,           // ← 标记为编辑中
                    editContent: t.content     // ← 复制内容到编辑缓冲区
                };
            }
            return { ...t, isEditing: editing };
        })
    );
}
```

```javascript
function updateEditContent(id: string, newContent: string) {
    tabs.update((ts) =>
        ts.map((t) => {
            if (t.id !== id) return t;
            return { 
                ...t, 
                editContent: newContent,                    // 更新编辑内容
                dirty: newContent !== t.content             // ★ 标记为"未保存"
            };
        })
    );
}
```

**此时状态**:
- `editContent` = 用户输入的新内容
- `content` = 原始文件内容（磁盘上的）
- `dirty` = true（有未保存更改）

## 第9步：handleSave 保存文件

**文件**: `src/routes/+page.svelte:211-241`
```javascript
async function handleSave(tab: Tab) {
    if (!tab.dirty) return;     // 没有更改就不保存
    
    try {
        // ① 调用 Rust 写入文件
        await saveFile(tab.filePath, tab.editContent);
        
        // ② 重新渲染（内容可能变了）
        const result = renderFull(tab.editContent, baseDir);
        
        // ③ 标记为已保存
        tabStore.markSaved(tab.id);
        
        // ④ 更新标签内容
        tabStore.updateTabContent(
            tab.filePath,
            tab.editContent,      // editContent 变成新的 content
            result.html,
            result.frontmatter,
            result.wordCount
        );
        
        // ⑤ 同步文档状态（用于显示）
        docStore.set({
            filePath: tab.filePath,
            content: tab.editContent,
            renderedHtml: result.html,
            ...
        });
        
    } catch (err) {
        alert(`Save failed: ${err}`);
    }
}
```

```rust
#[tauri::command]
pub fn write_markdown_file(path: String, content: String) -> Result<(), String> {
    let p = Path::new(&path);

    // 检查是否是文件（不是目录）
    if p.exists() && !p.is_file() {
        return Err(format!("Not a file: {}", path));
    }

    // ★ 写入文件
    fs::write(p, content).map_err(|e| format!("Failed to write: {}", e))
}
```

**返回**: Ok(()) 或 Err

```javascript
function markSaved(id: string) {
    tabs.update((ts) =>
        ts.map((t) => {
            if (t.id !== id) return t;
            return { 
                ...t, 
                content: t.editContent,     // ★ editContent → content
                dirty: false,               // ★ 清除脏标记
                lastSavedAt: Date.now()     // 记录保存时间
            };
        })
    );
}
```

**保存后状态**:
- `content` = editContent（新内容）
- `editContent` = 新内容
- `dirty` = false

## 数据状态变化

```
编辑前:
  content = "原始内容"
  editContent = "原始内容"
  dirty = false

编辑中:
  content = "原始内容"          (不变)
  editContent = "修改后内容"    (用户输入)
  dirty = true                  (有未保存更改)

保存后:
  content = "修改后内容"        (更新)
  editContent = "修改后内容"    (保持)
  dirty = false                 (已保存)
```
